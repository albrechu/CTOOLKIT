CMAKE_MINIMUM_REQUIRED(VERSION 3.14)

PROJECT(CTOOLKIT)

# ALLOCATOR
ADD_LIBRARY(TOOLKIT_ALLOCATOR STATIC SOURCE/ALLOCATOR.c)
TARGET_INCLUDE_DIRECTORIES(TOOLKIT_ALLOCATOR PUBLIC ${PROJECT_SOURCE_DIR}/INCLUDE)
SET_TARGET_PROPERTIES(TOOLKIT_ALLOCATOR PROPERTIES CXX_STANDARD 20 C_STANDARD 23 CXX_STANDARD_REQUIRED ON C_STANDARD_REQUIRED ON)
ADD_LIBRARY(TOOLKIT::ALLOCATOR ALIAS TOOLKIT_ALLOCATOR)
if(MSVC)
    TARGET_COMPILE_OPTIONS(TOOLKIT_ALLOCATOR PRIVATE "/Zc:__STDC__")
endif()

# NETWORK
ADD_LIBRARY(TOOLKIT_NETWORK STATIC SOURCE/NETWORK.c)
TARGET_INCLUDE_DIRECTORIES(TOOLKIT_NETWORK PUBLIC ${PROJECT_SOURCE_DIR}/INCLUDE)
ADD_LIBRARY(TOOLKIT::NETWORK ALIAS TOOLKIT_NETWORK)

# COMPRESSION
ADD_LIBRARY(TOOLKIT_COMPRESSION STATIC SOURCE/COMPRESSION.c)
TARGET_INCLUDE_DIRECTORIES(TOOLKIT_COMPRESSION PUBLIC ${PROJECT_SOURCE_DIR}/INCLUDE)
TARGET_LINK_LIBRARIES(TOOLKIT_COMPRESSION PUBLIC TOOLKIT::ALLOCATOR)
ADD_LIBRARY(TOOLKIT::COMPRESSION ALIAS TOOLKIT_COMPRESSION)

# FILESYSTEM
ADD_LIBRARY(TOOLKIT_FILESYSTEM STATIC SOURCE/FILESYSTEM.c)
TARGET_INCLUDE_DIRECTORIES(TOOLKIT_FILESYSTEM PUBLIC ${PROJECT_SOURCE_DIR}/INCLUDE)
TARGET_LINK_LIBRARIES(TOOLKIT_FILESYSTEM PUBLIC TOOLKIT::ALLOCATOR TOOLKIT::COMPRESSION)
ADD_LIBRARY(TOOLKIT::FILESYSTEM ALIAS TOOLKIT_FILESYSTEM)
SET_TARGET_PROPERTIES(TOOLKIT_FILESYSTEM PROPERTIES CXX_STANDARD 20 C_STANDARD 23 CXX_STANDARD_REQUIRED TRUE)

# FILE
ADD_LIBRARY(TOOLKIT_FILE STATIC SOURCE/FILE.c)
TARGET_INCLUDE_DIRECTORIES(TOOLKIT_FILE PUBLIC ${PROJECT_SOURCE_DIR}/INCLUDE)
TARGET_LINK_LIBRARIES(TOOLKIT_FILE PUBLIC TOOLKIT::ALLOCATOR TOOLKIT::COMPRESSION)
SET_TARGET_PROPERTIES(TOOLKIT_FILE PROPERTIES CXX_STANDARD 20 C_STANDARD 23 CXX_STANDARD_REQUIRED TRUE)
ADD_LIBRARY(TOOLKIT::FILE ALIAS TOOLKIT_FILE)

# COMMAND LINE INTERFACE
ADD_LIBRARY(TOOLKIT_CLI STATIC SOURCE/CLI.c)
TARGET_INCLUDE_DIRECTORIES(TOOLKIT_CLI PUBLIC ${PROJECT_SOURCE_DIR}/INCLUDE)
TARGET_LINK_LIBRARIES(TOOLKIT_CLI PUBLIC TOOLKIT::FILESYSTEM)
ADD_LIBRARY(TOOLKIT::CLI ALIAS TOOLKIT_CLI)
SET_TARGET_PROPERTIES(TOOLKIT_CLI PROPERTIES CXX_STANDARD 20 C_STANDARD 23 CXX_STANDARD_REQUIRED TRUE)

# THREAD
ADD_LIBRARY(TOOLKIT_THREAD STATIC SOURCE/THREAD.c)
TARGET_INCLUDE_DIRECTORIES(TOOLKIT_THREAD PUBLIC ${PROJECT_SOURCE_DIR}/INCLUDE)
TARGET_LINK_LIBRARIES(TOOLKIT_THREAD PUBLIC TOOLKIT::ALLOCATOR)
SET_TARGET_PROPERTIES(TOOLKIT_THREAD PROPERTIES CXX_STANDARD 20 C_STANDARD 23 CXX_STANDARD_REQUIRED ON C_STANDARD_REQUIRED ON)
ADD_LIBRARY(TOOLKIT::THREAD ALIAS TOOLKIT_THREAD)
if(MSVC)
    TARGET_COMPILE_OPTIONS(TOOLKIT_THREAD PRIVATE "/Zc:__STDC__" "/experimental:c11atomics")
endif()

# COMBINED
ADD_LIBRARY(TOOLKIT_TOOLKIT INTERFACE)
TARGET_LINK_LIBRARIES(TOOLKIT_TOOLKIT INTERFACE TOOLKIT::ALLOCATOR TOOLKIT::CLI TOOLKIT::COMPRESSION TOOLKIT::THREAD TOOLKIT::FILE TOOLKIT::FILESYSTEM TOOLKIT::NETWORK)
ADD_LIBRARY(TOOLKIT::TOOLKIT ALIAS TOOLKIT_TOOLKIT)

ADD_SUBDIRECTORY(EXAMPLES)
